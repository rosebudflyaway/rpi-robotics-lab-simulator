 6function sim = sphere_on_sphere()
 7 %#codegen
 8    % Initialize simulator
 9    sim = Simulator(0.01);
10    %sim.drawContacts = true;
11    sim.MAX_STEP = 500;     
12    sim.draw = false;
13    %sim.H_dynamics = @LCPdynamics; 
14    %sim.H_solver = @Lemke;
15    %sim.H_dynamics = @mLCPdynamics; 
16
17    % A sphere
18    s1 = Body_sphere(1,2);
19        s1.mass = 20;
20        s1.u = [0; 0; -1]; 
21        s1.color = [.7 .7 .7];
22        s1.num_sphere_verts = 25; 
23
24    s2 = Body_sphere(0.25,0.25);
25        s2.u = [-0.05; 0.05; 1.25];
26        s2.num_sphere_verts = 15;
27
28    P = Body_plane([0;0;-3],[0;0;1]); 
29        
30    sim = sim_addBody(sim, [P s1 s2]);    % Add bodies to simulator
31
32    % Run the simulator!
33    sim = sim_run(sim); 
34
35end
