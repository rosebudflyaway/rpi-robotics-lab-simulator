 5% Runs a simulator 
 6
 7function sim = sim_run( sim )
 8 %#codegen
 9
10    % Initialize graphics 
11    sim = sim_draw_init( sim );
12    if ~sim.draw, %disp('Simulating without GUI, make sure you have set sim.MAX_STEP to a reasonable value.'); end
13    
14    % Turn "on" gravity
15    %if sim.gravity, sim = sim_setGravity( sim ); end
16    
17    % Remove to run without pause
18    %pause(); 
19
20    %% Simulation loop 
21    %tic;
22    %sim.START_TIME = tic; 
23    for timestep = 1:sim.MAX_STEP 
24        %tic;                                % Start step timer
25        sim.time = sim.time + sim.h;
26        sim.step = sim.step + 1;
27        
28        % Run user-defined function: here it is wam_controller.m
29        if ~isempty(sim.userFunction), sim = feval( sim.userFunction, sim ); end
30        
31        % Generate a contact set, and identify participating dynamic bodies 
32%         if ~isempty( sim.H_collision_detection )
33%             sim = feval( sim.H_collision_detection, sim ); 
34%         end
35        
36        % Formulate dynamics and solve for new body velocities
37        %sim.newNU = [];
38       
39        numContact = length(sim.contacts); 
40        %fprintf('The number of contacts is: %6d\n', numContact);
41        
42        if ~isempty(sim.contacts) || sim.num_jointConstraints > 0  
43            sim = preDynamics( sim ); 
44            %sim = feval(sim.H_dynamics, sim);
45        end
46         
47         
48        % Apply results from solution, as well as update bodies not in contact.   
49        sim = body_updateAllBodies(sim);
50         
51        
52        % Correct joint errors
53        if sim.jointCorrection, sim = joint_correction( sim ); end
54        
55        % Update title and graphics
56%         if sim.draw
57%             figure(sim.figure);
58%             title(['Timestep: ' num2str(sim.step)]);
59%             if sim.drawContacts, sim = sim_drawContacts(sim); end
60%             if sim.drawJoints,   sim = sim_drawJoints(sim);   end
61%             axis equal;
62%             drawnow; 
63%         else
64%             %disp(['Step ' num2str(sim.step) ' took ' num2str(toc) ' seconds']);
65%         end
66        
67        % Record data for playback 
68        %if sim.record, sim = sim_record(sim); end 
69        
70    end % End simulation loop
71   % sim.TOTAL_TIME = toc(sim.START_TIME); 
72
73   % disp(['Simulation finished!  Total time: ' num2str(sim.TOTAL_TIME) ' seconds.']);
74    
75end % End sim_run()
