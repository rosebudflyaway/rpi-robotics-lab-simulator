 2% Creates a body struct with default body properties
 3
 4function B = Body()
 5 %#codegen
 6
 7    % The body struct
 8    %B = struct;
 9    
10    B.name = 'body';
11    B.type = 'generic';
12
13    B.bodyID = 0;           % Body's position in the simulation's vector of bodies
14    B.bodyContactID = 0;    % Body's id for current time-step contact set
15    B.bodyDynamicID = 0;    % TODO: duplicate of bodyContactID? 
16    
17    % Physical properties
18    B.dynamic = true;       % Dynamic vs. static body 
19    B.active = false;       % An active body is one that is in dynamic interaction with 
20                            %   another body through contact or some other constraint.  
21    B.mass = 1;             % Mass
22    B.u = zeros(3,1);       % Position of center of mass
23    B.quat = [1;0;0;0];     % Quaternion rotation
24    B.J = eye(3);           % Inertia tensor
25    %B.v = zeros(3,1);       % Linear velocity
26    %B.w = zeros(3,1);       % Angular velocity
27    B.nu = zeros(6,1);      % Generalized velocity [v; w]
28
29    B.Fext = zeros(6,1);    % External force
30    B.Aext = zeros(6,1);    % External acceleration
31    
32    B.mu = 0.5;             % Friction factor
33    
34    % Joint-related properties
35    B.numJoints = 0; 
36    
37    % Graphics properties 
38    B.visible = true; 
39    B.graphicsHandle = [];  % Graphics object for plotting the body
40    B.Xdata = [];
41    B.Ydata = [];
42    B.Zdata = [];
43    B.color = [0 0 1];      % Color of body
44    B.facealpha = 0.7;
45    B.edgealpha = 1; 
46    B.bboxHandle = [];      % Graphics object for plotting bounding box
47
48    % Collision detection properties
49    B.AABB_min = zeros(1,3); % Bounding box
50    B.AABB_max = zeros(1,3); 
51
52    B.collides = true;         % If false, this body won't collide with any other bodies
53    B.doesNotCollideWith = []; % A list of bodyIDs of bodies not to do collision detection with
54    
55    
56    %% Different bodies have different attributes, but for simplicity 
57    %  we want all bodies to be represented by this struct.  Therefore we
58    %  place all body attributes here, and let bodies pick and choose.
59    
60    %% Sphere attributes
61    B.radius = 1;
62    B.num_sphere_verts = 7;  % Decrease to improve graphics performance 
63                             % (does not affect simulation result)
64    
65    %% Cylinder attributes
66    B.height = 1; 
67                             
68    %% Plane attributes
69    B.plane_normal = [0 0 1];
70    
71    %% Mesh attributes
72    B.num_verts = 0;
73    B.num_edges = 0;
74    B.num_faces = 0; 
75
76    B.verts_local = [];  % A Nx3 vector of N vertices relative to the COM of the body
77    B.verts_world = [];  % A Nx3 vector of N vertices in the world space
78    B.edges = [];        % A Ex [v1 v2 Enext Eopp face]   TODO: remove implicit columns Enext and face
79                            % A Ex5 vector [v1 v2 f1 f2 a] where v1:first
80                            % vert, v2: second vert, f1: first face, f2:
81                            % second face, a: angle between faces
82    B.tvecs = [];        % A Ex6 vector of t vectors (always in local coords), perp to edge and planar to adjacent faces
83    B.faces = [];        % A Mx4 vector of M faces where each row contains: [v1 v2 v3 e1]
84    B.face_norms = [];   % A Mx3 vector of M face norms where each row contains [n1 n2 n3] 
85    
86    
87end
